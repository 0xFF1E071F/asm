object Macro f,p:VARARG
    LOCAL obj
    LOCAL poz
    LOCAL txt, arg
    txt TEXTEQU <>
    FOR arg, <p>
        txt CATSTR <arg>, <!,>, txt
    ENDM
    txt SUBSTR  txt, 1, @SizeStr( %txt ) - 1
    txt CATSTR  <!<>, txt, <!>>
	
    % poz INSTR 1,<f>,<.>
    obj SUBSTR <f>, 1, poz-1

    % FOR arg, txt
        push arg
    ENDM

    push offset obj
    call f
endm
